const testAuth = async () => {
  console.log('=== Testing Authentication ===\n');

  try {
    const payload = JSON.parse(atob(token.split('.')[1]));

    console.log('2. Token expiry:', expiry.toLocaleString());
    console.log('   Is expired:', isExpired);
    
    if (isExpired) {
      console.log('   ‚Üí TOKEN EXPIRED');
      console.log('   ‚Üí Action: Login again');
      return;
    }

    console.log('   ‚Üí TOKEN INVALID (cannot parse)');
    console.log('   ‚Üí Action: Clear storage and login');
    return;
  }
  
  // 3. Test endpoints
  console.log('\n3. Testing endpoints...');

    const res = await fetch('/api/auth/navbar-info', {

      console.log('   ‚Üí User:', data.name, `(${data.role})`);
    }
  } catch (e) {
    console.log('   navbar-info: ERROR', e.message);
  }
  
  console.log('\n‚úÖ Test complete!');
